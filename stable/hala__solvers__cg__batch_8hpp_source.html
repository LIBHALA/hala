67 template<class VectorLikeP, class VectorLikeI, class VectorLikeV, class VectorLikeX, class VectorLikeB>
68 int solve_batch_cg(stop_criteria<get_precision_type<VectorLikeV>> const &stop,
69  VectorLikeP const &pntr, VectorLikeI const &indx, VectorLikeV const &vals,
70  preconditioner_noe<VectorLikeV> precon,
71  VectorLikeB const &B, VectorLikeX &X){
72 
73  check_types(vals, X, B);
74  check_types_int(pntr, indx);
75 
76  int num_entries = get_size_int(pntr);
77  assert( num_entries > 1 ); // smallest case has one entry
78  int num_rhs = get_size_int(B) / (num_entries - 1);
79 
80  if (get_size(X) < get_size(B)){
81  force_size(get_size(B), X);
82  set_zero(get_size(B), X); // if defined on CUDA the entries will not be initialized
83  }
84 
85  using temp_vector = decltype( new_vector( std::declval<VectorLikeV>() ) );
86  using scalar_type = temp_vector;
87 
88  int nrows = get_size_int(pntr) - 1;
89 
90  auto matrix = make_sparse_matrix(nrows, pntr, indx, vals);
91  size_t bsize = sparse_gemm_buffer_size(matrix, 'N', 'N', nrows, num_rhs, 1.0, B, nrows, 0.0, X, nrows);
92  auto work_buffer = new_vector(X);
93  set_size(bsize, work_buffer);
94 
95  return solve_cg_core<scalar_type>(precon,
96  [&](VectorLikeX const &v, temp_vector &r)->void{
97  sparse_gemm(matrix, 'N', 'N', nrows, num_rhs, 1.0, v, nrows, 0.0, r, nrows, work_buffer);
98  },
99  [&](temp_vector const &v, temp_vector &r)->void{
100  sparse_gemm(matrix, 'N', 'N', nrows, num_rhs, 1.0, v, nrows, 0.0, r, nrows, work_buffer);
101  },
102  B, X,
103  [&]()->scalar_type{ return new_vector(X); },
104  [&](scalar_type const &num, scalar_type const &denom, scalar_type &result)->void{ vdivide(num, denom, result); },
105  [](scalar_type &src, scalar_type &dest)->void{ dest = std::move(src); },
106  [&](scalar_type const &a, temp_vector &r)->void{ batch_scal(num_rhs, a, r); },
107  [&](temp_vector const &v, temp_vector &r)->void{ axpy( 1.0, v, r); },
108  [&](temp_vector const &v, temp_vector &r)->void{ axpy(-1.0, v, r); },
109  [&](scalar_type const &a, temp_vector const &v, VectorLikeX &r)->void{ batch_axpy(num_rhs, 1.0, a, v, r); },
110  [&](scalar_type const &a, temp_vector const &v, temp_vector &r)->void{ batch_axpy(num_rhs, -1.0, a, v, r); },
111  [&](temp_vector const &v, temp_vector const &r, scalar_type &a)->void{ batch_dot(num_rhs, v, r, a); },
112  [&](int i, temp_vector const &r)->bool{ return ((i == stop.max_iter) || (batch_max_norm2(num_rhs, r) < stop.tol)); }
113  );
114 }
