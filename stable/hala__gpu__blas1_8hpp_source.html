269 template<typename FC, typename FS, class VectorLikeX, class VectorLikeY>
270 void rot(gpu_engine const &engine, int N, VectorLikeX &x, int incx, VectorLikeY &&y, int incy, FC C, FS S){
271  check_types(x, y); // make sure x and y types match
272  engine.check_gpu(x, y);
273  assert( valid::rot(N, x, incx, y, incy) );
274 
275  using scalar_type = get_scalar_type<VectorLikeX>;
276  using precision_type = get_precision_type<VectorLikeX>;
277 
278  #ifdef HALA_ENABLE_CUDA
279  if __HALA_CONSTEXPR_IF__ ((is_fcomplex<scalar_type>::value || is_dcomplex<scalar_type>::value) &&
280  (std::is_same<FS, int>::value || is_float<FS>::value || is_double<FS>::value)){ // mix complex and real
281  auto effc = get_pointer<precision_type>(C);
282  auto effs = get_pointer<precision_type>(S);
283  if __HALA_CONSTEXPR_IF__ (is_fcomplex<scalar_type>::value){
284  cublasCsrot(engine, N, cconvert(x), incx, cconvert(y), incy, effc, effs);
285  }else{
286  cublasZdrot(engine, N, cconvert(x), incx, cconvert(y), incy, effc, effs);
287  }
288  }else{ // scaled by the same type
289  auto effc = get_pointer<precision_type>(C);
290  auto effs = get_pointer<scalar_type>(S);
291  cuda_call_backend<scalar_type>(cublasSrot, cublasDrot, cublasCrot, cublasZrot,
292  "GPU-BLAS::Xrot()", engine, N, cconvert(x), incx, cconvert(y), incy, effc, effs);
293  }
294  #endif
295  #ifdef HALA_ENABLE_ROCM
296  if __HALA_CONSTEXPR_IF__ ((is_fcomplex<scalar_type>::value || is_dcomplex<scalar_type>::value) &&
297  (std::is_same<FS, int>::value || is_float<FS>::value || is_double<FS>::value)){ // mix complex and real
298  auto effc = get_pointer<precision_type>(C);
299  auto effs = get_pointer<precision_type>(S);
300  if __HALA_CONSTEXPR_IF__ (is_fcomplex<scalar_type>::value){
301  rocblas_csrot(engine, N, cconvert(x), incx, cconvert(y), incy, effc, effs);
302  }else{
303  rocblas_zdrot(engine, N, cconvert(x), incx, cconvert(y), incy, effc, effs);
304  }
305  }else{ // scaled by the same type
306  auto effc = get_pointer<precision_type>(C);
307  auto effs = get_pointer<scalar_type>(S);
308  rocm_call_backend<scalar_type>(rocblas_srot, rocblas_drot, rocblas_crot, rocblas_zrot,
309  "rocBlas::Xrot()", engine, N, cconvert(x), incx, cconvert(y), incy, effc, effs);
310  }
311  #endif
