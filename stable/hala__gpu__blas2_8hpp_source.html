171 template<bool conjugate = false, typename FSa, typename FSb, class VectorLikeA, class VectorLikeX, class VectorLikeY>
172 inline void spmv(gpu_engine const &engine, char uplo, int N, FSa alpha, const VectorLikeA &A,
173  const VectorLikeX &x, int incx, FSb beta, VectorLikeY &&y, int incy){
174  check_types(A, x, y);
175  engine.check_gpu(A, x, y);
176  pntr_check_set_size(beta, y, 1 + incy * (N - 1), 1);
177  assert( valid::spmv(uplo, N, A, x, incx, y, incy) );
178 
179  using scalar_type = get_scalar_type<VectorLikeA>;
180 
181  #ifdef HALA_ENABLE_CUDA
182  if __HALA_CONSTEXPR_IF__ (conjugate || !is_complex<scalar_type>::value){
183  cublasFillMode_t cuda_uplo = uplo_to_gpu(uplo);
184 
185  auto palpha = get_pointer<scalar_type>(alpha);
186  auto pbeta = get_pointer<scalar_type>(beta);
187 
188  cuda_call_backend<scalar_type>(cublasSspmv, cublasDspmv, cublasChpmv, cublasZhpmv,
189  "GPU-BLAS::Xspmv()", engine, cuda_uplo, N, palpha, convert(A), convert(x), incx, pbeta, cconvert(y), incy);
190  }else{
191  // since Cspmv and Zspmv are missing from the GPU-BLAS standard, unpack A, form full matrix, use gemv
192  auto full_matrix = new_vector(engine, A);
193  { // respect the user selected pointer mode
194  gpu_pntr<host_pntr> hold(engine);
195  auto unpacked = new_vector(engine, A);
196  force_size(hala_size(N, N), unpacked);
197  scal(engine, 0.0, unpacked);
198  tp2tr(engine, uplo, N, A, unpacked, N);
199  force_size(hala_size(N, N), full_matrix);
200  scal(engine, 0.0, full_matrix);
201  geam(engine, 'N', 'T', N, N, 1.0, unpacked, N, 1.0, unpacked, N, full_matrix, N);
202  scal(engine, N, 0.5, full_matrix, N+1);
203  }
204  gemv(engine, 'N', N, N, alpha, full_matrix, N, x, incx, beta, y, incy);
205  }
206  #endif
207  #ifdef HALA_ENABLE_ROCM
208  if __HALA_CONSTEXPR_IF__ (conjugate || !is_complex<scalar_type>::value){
209  auto cuda_uplo = uplo_to_gpu(uplo);
210 
211  auto palpha = get_pointer<scalar_type>(alpha);
212  auto pbeta = get_pointer<scalar_type>(beta);
213 
214  rocm_call_backend<scalar_type>(rocblas_sspmv, rocblas_dspmv, rocblas_chpmv, rocblas_zhpmv,
215  "rocBlas::Xspmv()", engine, cuda_uplo, N, palpha, convert(A), convert(x), incx, pbeta, cconvert(y), incy);
216  }else{
217  // fallback, since cspmv and zspmv are missing from the standard, use the CPU
218  auto cA = engine.unload(A);
219  auto cx = engine.unload(x);
220  auto cy = engine.unload(y);
221  spmv<conjugate>(uplo, N, alpha, cA, cx, incx, beta, cy, incy);
222  gpu_copy_n<copy_direction::host2device>(get_data(cy), get_size(y), get_data(y));
223  }
224  #endif
225 }
