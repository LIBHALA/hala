394 template<class VectorLike>
395 void set_zero(gpu_engine const &engine, size_t num_entries, VectorLike &&x){
396  if (num_entries == 0) return;
397  check_set_size(assume_output, x, num_entries);
398  engine.set_active_device();
399 
400  using standard_type = get_standard_type<VectorLike>;
401  auto data = get_standard_data(x);
402 
403  standard_type value = get_cast<standard_type>(0.0);
404  gpu_copy_n<copy_direction::host2device>(&value, 1, data);
405 
406  size_t loaded = 1;
407  while(loaded < num_entries){
408  gpu_copy_n<copy_direction::device2device>(data, std::min(loaded, num_entries - loaded), data + loaded);
409  loaded *= 2;
410  }
411 }
