181  template<typename FSA, class VectorLikeB, typename FSB, class VectorLikeC, class VectorLikeT>
182  void gemm(char transa, char transb, int b_rows, int b_cols, FSA alpha, VectorLikeB const &B, int ldb, FSB beta, VectorLikeC &C, int ldc, VectorLikeT &&temp) const{
183  check_types(rvals, B, C);
184  rengine.check_gpu(B, C);
185  int M = (is_n(transa)) ? rows : cols;
186  int N = (is_n(transb)) ? b_cols : b_rows;
187  int K = (is_n(transa)) ? cols : rows;
188  pntr_check_set_size(beta, C, ldc, N);
189  assert( valid::sparse_gemm(transa, transb, M, N, K, nnz, rpntr, rindx, rvals, B, ldb, C, ldc) );
190 
191  auto palpha = get_pointer<value_type>(alpha);
192  auto pbeta = get_pointer<value_type>(beta);
193 
194  auto rocm_transa = trans_to_rocm_sparse<value_type>(transa);
195  auto rocm_transb = trans_to_rocm_sparse<value_type>(transb);
196  runtime_assert(rocm_transa == rocsparse_operation_none, "as of rocsparse 3.5 there is no support for transpose and conjugate-transpose gemm operations.");
197 
198  if (is_complex<value_type>::value and rocm_transb == rocsparse_operation_conjugate_transpose){
199  // employing manual out-of-place conj-transpose for B
200  geam(rengine, 'C', 'C', K, N, 1.0, B, ldb, 0.0, B, ldb, temp, K);
201  rocm_call_backend<value_type>(rocsparse_scsrmm, rocsparse_dcsrmm, rocsparse_ccsrmm, rocsparse_zcsrmm,
202  "csrmm()", rengine, rocm_transa, rocsparse_operation_none, M, N, K, nnz, palpha, desc, convert(rvals), convert(rpntr), convert(rindx), convert(temp), K, pbeta, cconvert(C), ldc);
203  }else{
204  rocm_call_backend<value_type>(rocsparse_scsrmm, rocsparse_dcsrmm, rocsparse_ccsrmm, rocsparse_zcsrmm,
205  "csrmm()", rengine, rocm_transa, rocm_transb, M, N, K, nnz, palpha, desc, convert(rvals), convert(rpntr), convert(rindx), convert(B), ldb, pbeta, cconvert(C), ldc);
206  }
