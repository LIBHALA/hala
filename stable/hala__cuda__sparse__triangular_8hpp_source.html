293  template<typename FPA, class VectorLikeB, class VectorLikeT>
294  void trsm(char transa, char transb, int nrhs, FPA alpha, VectorLikeB &&B, int ldb, VectorLikeT &&) const{
295  check_types(rvals, B);
296  valid::default_ld(is_n(transb), nrows, nrhs, ldb);
297  assert( valid::sparse_trsm(transa, transb, nrhs, *this, B, ldb) );
298 
299  auto palpha = get_pointer<value_type>(alpha);
300 
301  auto cuda_transa = trans_to_cuda_sparse<value_type>(transa);
302  if (is_c(transb)) transb = 'T'; // the conj operation cancels out, use simple transpose
303 
304  auto C = new_vector(rengine, B);
305  force_size(hala_size(nrows, nrhs), C);
306 
307  if (is_n(transb)){
308 
309  cuda_call_backend<value_type>(cusparseScsrsm_solve, cusparseDcsrsm_solve, cusparseCcsrsm_solve, cusparseZcsrsm_solve,
310  "cusparseXcsrsm_solve", rengine, cuda_transa, nrows, nrhs, palpha, cdesc,
311  vals(), rpntr, rindx, analyze_sv(transa), convert(B), ldb, cconvert(C), nrows);
312 
313  if (ldb == nrows){
314  vcopy(rengine, C, B); // copy back
315  }else{ // copy only the block
316  gpu_pntr<host_pntr> hold(rengine); // in case device pointers were used in the call above
317  geam(rengine, 'N', 'N', nrows, nrhs, 1, C, nrows, 0, C, nrows, B, ldb);
318  }
319  }else{
320 
321  auto X = new_vector(rengine, B); // set X = B^T
322  force_size(hala_size(nrows, nrhs), X);
323  {
324  gpu_pntr<host_pntr> hold(rengine);
325  geam(rengine, 'T', 'T', nrows, nrhs, 1, B, ldb, 0, B, ldb, X, nrows);
326  }
327 
328  cuda_call_backend<value_type>(cusparseScsrsm_solve, cusparseDcsrsm_solve, cusparseCcsrsm_solve, cusparseZcsrsm_solve,
329  "cusparseXcsrsm_solve", rengine, cuda_transa, nrows, nrhs, palpha, cdesc,
330  vals(), rpntr, rindx, analyze_sv(transa), convert(X), nrows, cconvert(C), nrows);
331 
332  { // C holds the answer, write back to B in transposed order
333  gpu_pntr<host_pntr> hold(rengine);
334  geam(rengine, 'T', 'T', nrhs, nrows, 1, C, nrows, 0, C, nrows, B, ldb);
335  }
336 
337  }
