103 template<typename T, char trans>
104 inline void sparse_gemv_array(int num_rows, int num_cols, T alpha, int const pntr[], int const indx[], T const vals[], T const x[], T beta, T y[]){
105  if __HALA_CONSTEXPR_IF__ (trans == 'N'){
106  for(int i=0; i<num_rows; i++){
107  T sum = get_cast<T>(0.0);
108  for(int j=pntr[i]; j<pntr[i+1]; j++)
109  sum += vals[j] * x[ indx[j] ];
110  y[i] = alpha * sum + beta * y[i];
111  }
112  }else{
113  scal(num_cols, beta, y, 1);
114  for(int i=0; i<num_rows; i++)
115  for(int j=pntr[i]; j<pntr[i+1]; j++)
116  y[indx[j]] += alpha * x[i] * ((trans == 'C') ? cconj(vals[j]) : vals[j]);
117  }
118 }

175 template<bool set_pntrs, typename T = int>
176 void split_pattern_array(T num_rows, T const pntr[], T const indx[], T const diag[],
177  T upper_pntr[], T upper_indx[], T lower_pntr[], T lower_indx[]){
178  if (set_pntrs){
179  upper_pntr[0] = 0;
180  lower_pntr[0] = 0;
181 
182  for(T i=0; i<num_rows; i++){
183  lower_pntr[i+1] = lower_pntr[i] + diag[i] - pntr[i] + 1;
184  upper_pntr[i+1] = upper_pntr[i] + pntr[i+1] - diag[i];
185  }
186  }else{
187  for(T i=0; i<num_rows; i++){
188  std::copy_n(indx + pntr[i], lower_pntr[i+1] - lower_pntr[i], lower_indx + lower_pntr[i]);
189  std::copy_n(indx + diag[i], upper_pntr[i+1] - upper_pntr[i], upper_indx + upper_pntr[i]);
190  }

201 void split_values_array(Tindex num_rows, char uplo, char diag,
202  Tindex const pntr[], T const vals[],
203  Tindex const upper_pntr[], Tindex const lower_pntr[],
204  T upper_vals[], T lower_vals[]){
205 
206  T alpha = is_n(diag) ? get_cast<T>(0.0) : get_cast<T>(1.0);
207 
208  for(Tindex i=0; i<num_rows; i++){
209  Tindex num_lower = lower_pntr[i+1] - lower_pntr[i];
210  std::copy_n(vals + pntr[i], num_lower, lower_vals + lower_pntr[i]);
211  std::copy_n(vals + pntr[i] + num_lower -1,
212  upper_pntr[i+1] - upper_pntr[i],
213  upper_vals + upper_pntr[i]);
214 
215  if (is_l(uplo)) // set upper entry to alpha
216  upper_vals[upper_pntr[i]] = alpha;
217  else // set the lower entry to alpha
218  lower_vals[lower_pntr[i+1] - 1] = alpha;
219  }

228 template<typename T>
229 inline void factorize_ilu_array(int num_rows, int const pntr[], int const indx[], int const diag[], T ilu[]){
230  for(int i=0; i<num_rows; i++){
231  auto u = ilu[diag[i]];
232  //#pragma omp parallel for
233  for(int j=i+1; j<num_rows; j++){
234  auto jc = pntr[j];
235  while(indx[jc] < i) jc++;
236  if (indx[jc] == i){
237  ilu[jc] /= u;
238  auto l = ilu[jc];
239  auto ik = diag[i]+1;
240  auto jk = jc+1;
241  while((ik<pntr[i+1]) && (jk<pntr[j+1])){
242  if (indx[ik] == indx[jk]){
243  ilu[jk] -= l * ilu[ik];
244  ik++; jk++;
245  }else if (indx[ik] < indx[jk]){
246  ik++;
247  }else{
248  jk++;
249  }
250  }
251  }
252  }
253  }

262 template<typename T>
263 inline void apply_ilu_array(int num_rows, int const pntr[], int const indx[], int const diag[], T const ilu[], T x[]){
264  // action of the preconditioner
265  for(int i=1; i<num_rows; i++)
266  for(int j=pntr[i]; j<diag[i]; j++)
267  x[i] -= ilu[j] * x[indx[j]];
268 
269  for(int i=num_rows-1; i>=0; i--){
270  for(int j=diag[i]+1; j<pntr[i+1]; j++)
271  x[i] -= ilu[j] * x[indx[j]];
272 
273  x[i] /= ilu[diag[i]];
274  }
275 }

283 template<char diag, typename T, typename Tindex = int>
284 void sparse_trsv_array(char uplo, char trans, Tindex num_rows,
285  T alpha, Tindex const pntr[], Tindex const indx[], T const vals[], T const b[], T x[]){
286  if (is_n(trans)){
287  if (uplo == 'L'){
288  x[0] = (diag == 'U') ? alpha * b[0] : alpha * b[0] / vals[0];
289  for(int i=1; i<num_rows; i++){
290  T s = get_cast<T>(0.0);
291  for(int j=pntr[i]; j<pntr[i+1]-1; j++)
292  s += vals[j] * x[indx[j]];
293  x[i] = (diag == 'U') ? (alpha * b[i] - s) : (alpha * b[i] - s) / vals[ pntr[i+1]-1 ];
294  }
295  }else{
296  x[num_rows - 1] = (diag == 'U') ? alpha * b[num_rows - 1] : alpha * b[num_rows - 1] / vals[ pntr[num_rows] -1 ];
297  for(int i=num_rows-2; i>=0; i--){
298  T s = get_cast<T>(0.0);
299  for(int j=pntr[i]+1; j<pntr[i+1]; j++)
300  s += vals[j] * x[indx[j]];
301  x[i] = (diag == 'U') ? (alpha * b[i] - s) : (alpha * b[i] - s) / vals[ pntr[i] ];
302  }
303  }
304  }else if (is_c(trans)){ // conjugate transpose
305  hala::scal(num_rows, 0.0, x, 1);
306  if (uplo == 'L'){
307  for(int i=num_rows-1; i>=0; i--){
308  x[i] = (diag == 'U') ? (alpha * b[i] - x[i]) : (alpha * b[i] - x[i]) / hala::cconj(vals[ pntr[i+1] - 1 ]);
309  for(int j=pntr[i]; j<pntr[i+1] - 1; j++) // last entry is the diagonal, do not update
310  x[indx[j]] += x[i] * hala::cconj(vals[j]);
311  }
312  }else{
313  for(int i=0; i<num_rows; i++){
314  x[i] = (diag == 'U') ? (alpha * b[i] - x[i]) : (alpha * b[i] - x[i]) / hala::cconj(vals[ pntr[i] ]);
315  for(int j=pntr[i] + 1; j<pntr[i+1]; j++)
316  x[indx[j]] += x[i] * hala::cconj(vals[j]);
317  }
318  }
319  }else{ // non-conj transpose
320  hala::scal(num_rows, 0.0, x, 1);
321  if (uplo == 'L'){
322  for(int i=num_rows-1; i>=0; i--){
323  x[i] = (diag == 'U') ? (alpha * b[i] - x[i]) : (alpha * b[i] - x[i]) / vals[ pntr[i+1] - 1 ];
324  for(int j=pntr[i]; j<pntr[i+1] - 1; j++) // last entry is the diagonal, do not update
325  x[indx[j]] += x[i] * vals[j];
326  }
327  }else{
328  for(int i=0; i<num_rows; i++){
329  x[i] = (diag == 'U') ? (alpha * b[i] - x[i]) : (alpha * b[i] - x[i]) / vals[ pntr[i] ];
330  for(int j=pntr[i] + 1; j<pntr[i+1]; j++)
331  x[indx[j]] += x[i] * vals[j];
332  }
333  }
334  }
335 }
