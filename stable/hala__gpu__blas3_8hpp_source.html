39 template<typename FS, class VectorLikeA, class VectorLikeB>
40 inline void trmm(gpu_engine const &engine, char side, char uplo, char trans, char diag, int M, int N, FS alpha, const VectorLikeA &A, int lda, VectorLikeB &&B, int ldb){
41  check_types(A, B);
42  engine.check_gpu(A, B);
43  assert( valid::trmsm(side, uplo, trans, diag, M, N, A, lda, B, ldb) );
44 
45  using scalar_type = get_scalar_type<VectorLikeA>;
46  auto palpha = get_pointer<scalar_type>(alpha);
47 
48  auto gpu_side = side_to_gpu(side);
49  auto gpu_uplo = uplo_to_gpu(uplo);
50  auto gpu_trans = trans_to_gpu(trans);
51  auto gpu_diag = diag_to_gpu(diag);
52 
53  #ifdef HALA_ENABLE_CUDA
54  auto C = new_vector(engine, B);
55  force_size(hala_size(M, N), C);
56 
57  cuda_call_backend<scalar_type>(cublasStrmm, cublasDtrmm, cublasCtrmm, cublasZtrmm,
58  "GPU-BLAS::Xtrmm()", engine, gpu_side, gpu_uplo, gpu_trans, gpu_diag,
59  M, N, palpha, convert(A), lda, convert(B), ldb, cconvert(C), M);
60  if (ldb == M){
61  vcopy(engine, C, B); // copy back
62  }else{ // copy only the block
63  gpu_pntr<host_pntr> hold(engine); // in case device pointers were used in the call above
64  geam(engine, 'N', 'N', M, N, 1, C, M, 0, C, M, B, ldb);
65  }
66  #endif
67  #ifdef HALA_ENABLE_ROCM
68  rocm_call_backend<scalar_type>(rocblas_strmm, rocblas_dtrmm, rocblas_ctrmm, rocblas_ztrmm,
69  "rocBlas::Xtrmm()", engine, gpu_side, gpu_uplo, gpu_trans, gpu_diag,
70  M, N, palpha, convert(A), lda, convert(B), ldb);
71  #endif
72 }
