120 inline auto make_cuda_mat_description(char type, char uplo, char diag){
121  cuda_struct_description<cusparseMatDescr_t> result;
122  check_cuda( cusparseCreateMatDescr(&result.desc), "cuSparse::CreateMatDescr()");
123  cusparseSetMatType(result.desc, (type == 'G') ? CUSPARSE_MATRIX_TYPE_GENERAL : CUSPARSE_MATRIX_TYPE_TRIANGULAR);
124  cusparseSetMatIndexBase(result.desc, CUSPARSE_INDEX_BASE_ZERO);
125  if (check_diag(diag))
126  cusparseSetMatDiagType(result.desc, (is_n(diag)) ? CUSPARSE_DIAG_TYPE_NON_UNIT : CUSPARSE_DIAG_TYPE_UNIT);
127  if (check_uplo(uplo))
128  cusparseSetMatFillMode(result.desc, (is_l(uplo)) ? CUSPARSE_FILL_MODE_LOWER : CUSPARSE_FILL_MODE_UPPER);
129  return result;
