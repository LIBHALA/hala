182 struct aligned_iterator{

186  using reference = typename aligned_iterator_traits<T, R, vec>::reference;

192  using iterator_category = std::random_access_iterator_tag;
194  iterator i;
195

199  aligned_iterator(iterator x) : i(x){}
200 
202  reference operator *() const{ return *i; }
203 
205  aligned_iterator<T, R, vec>& operator ++(){
206  i += vex_stride<T, R>::stride;
207  return *this;
208  }
210  aligned_iterator<T, R, vec>& operator --(){
211  i -= vex_stride<T, R>::stride;
212  return *this;
213  }
215  aligned_iterator<T, R, vec> operator ++(int){
216  aligned_iterator<T, R, vec> t(*this);
217  i += vex_stride<T, R>::stride;
218  return t;
219  }
221  aligned_iterator<T, R, vec> operator --(int){
222  aligned_iterator<T, R, vec> t(*this);
223  i -= vex_stride<T, R>::stride;
224  return t;
225  }
226 
228  aligned_iterator<T, R, vec> operator += (difference_type const n){
229  i += n * vex_stride<T, R>::stride;
230  return *this;
