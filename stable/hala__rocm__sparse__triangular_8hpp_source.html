104  void trsv(char trans, FPA alpha, VectorLikeB const &b, VectorLikeX &&x, VectorLikeT &&temp) const{
105  check_types(rvals, b, x);
106  assert( valid::sparse_trsv(trans, *this, b) );
107  check_set_size(assume_output, x, nrows);
108 
109  auto rocm_trans = trans_to_rocm_sparse<value_type>(trans);
110  auto palpha = get_pointer<value_type>(alpha);
111 
112  runtime_assert(rocm_trans != rocsparse_operation_conjugate_transpose, "as of rocsparse 3.5 there is no support for conjugate-transpose trsv operations.");
113 
114  rocm_call_backend<value_type>(rocsparse_scsrsv_solve, rocsparse_dcsrsv_solve, rocsparse_ccsrsv_solve, rocsparse_zcsrsv_solve,
115  "csrsv_solve", rengine, rocm_trans, nrows, nz, palpha, cdesc.desc, vals(), rpntr, rindx,
116  analysis(rocm_trans, temp), convert(b), cconvert(x), rocsparse_solve_policy_auto, cconvert(temp));
117  }
