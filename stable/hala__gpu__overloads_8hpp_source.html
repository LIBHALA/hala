54 template<class vec>
55 auto gpu_engine::vcopy(vec const &x) const{
56  auto y = new_vector(*this, x);
57  auto xdevice = get_device(x);
58  if ((xdevice > -1) && (xdevice != cgpu)){
59  using standard_type = get_standard_type<vec>;
60  std::vector<standard_type> cpuy(get_size(x));
61  gpu_copy_n<copy_direction::device2host>(get_data(x), get_size(x), get_data(cpuy));
62  y.load(cpuy);
63  }else{
64  hala::vcopy(*this, x, y);
65  }
66  return y;
67 }
