125 template<class compute_engine, class VectorLikeP, class VectorLikeI, class VectorLikeV, class VectorLikeX, class VectorLikeB>
126 hala_int<compute_engine>
127 solve_gmres(compute_engine const &engine,
128  stop_criteria<get_precision_type<VectorLikeV>> const &stop, int restart,
129  VectorLikeP const &pntr, VectorLikeI const &indx, VectorLikeV const &vals,
130  preconditioner<compute_engine, VectorLikeV> precon,
131  VectorLikeB const &b, VectorLikeX &x){
132  // sanity check
133  check_types(vals, x, b);
134  check_types_int(pntr, indx);
135 
136  int num_rows = get_size_int(pntr) -1;
137  assert( num_rows > 0 ); // smallest case has one entry
138  assert( check_size(b, num_rows) );
139 
140  if (get_size_int(x) < num_rows){
141  force_size(num_rows, x);
142  set_zero(engine, (size_t) num_rows, x);
143  }
144 
145  // define the types
146  using standart_type = get_standard_type<VectorLikeV>;
147  using precision_type = get_precision_type<VectorLikeV>;
148 
149  std::vector<standart_type> H, S, Z;
150  std::vector<precision_type> C;
151  H.reserve(restart * (restart + 1)); // transformation matrix, upper triangular packed
152  S.reserve(restart + 1); // sin of Givens rotation
153  C.reserve(restart + 1); // cos of Givens rotation
154  Z.reserve(restart + 1); // holds the coefficients of the solution
155 
156  precision_type inner_res, outer_res = get_cast<precision_type>(stop.tol + 1.0); // inner and outer residuals
157  int total_iterations = 0;
158  int outer_iterations = 0;
159 
160  auto matrix = make_sparse_matrix(engine, num_rows, pntr, indx, vals);
161  size_t bsize = sparse_gemv_buffer_size(matrix, 'N', 1.0, b, 1.0, x);
162  auto work_buffer = new_vector(x);
163  set_size(bsize, work_buffer);
164 
165  while((outer_res > stop.tol) && (outer_iterations < stop.max_iter)){
166  H.resize(0); S.resize(0); C.resize(0); Z.resize(0);
167 
168  auto t = new_vector(engine, vals);
169  auto r = new_vector(engine, vals);
170 
171  vcopy(engine, b, t);
172  matrix.gemv('N', -1.0, x, 1.0, t, work_buffer);
173  precon(t, r);
174  total_iterations++;
175 
176  inner_res = norm2(engine, r);
177  scal(engine, 1.0 / inner_res, r);
178  Z.push_back(get_cast<standart_type>(inner_res));
179 
180  auto W = new_vector(engine, vals);
181  force_size(hala_size(num_rows, restart), W);
182  vcopy(engine, r, W);
183 
184  int inner_iterations = 0;
185  while ((inner_res > stop.tol) && (inner_iterations < restart)){
186 
187  matrix.gemv('N', 1.0, r, 0.0, t, work_buffer);
188  precon(t, r);
189  total_iterations++;
190 
191  std::vector<standart_type> coeffs;
192  krylov_project(engine, num_rows, inner_iterations + 1, r, W, coeffs);
193  auto nrm = norm2(engine, r);
194  scal(engine, 1.0 / nrm, r);
195 
196  for(int i=0; i<inner_iterations; i++)
197  rot(1, &coeffs[i], 1, &coeffs[i+1], 1, C[i], S[i]);
198 
199  standart_type isin, beta = get_cast<standart_type>(nrm);
200  precision_type icos;
201  rotg(coeffs[inner_iterations], beta, icos, isin);
202 
203  H.insert(H.end(), coeffs.begin(), coeffs.end());
204  S.push_back(isin);
205  C.push_back(icos);
206 
207  inner_res = hala_abs(S.back() * Z.back());
208  inner_iterations++;
209 
210  if ((inner_res > stop.tol) && (inner_iterations < restart)){ // if still going ...
211  // expand the Krylov basis
212  vcopy(engine, num_rows, r, 1, get_data(W) + inner_iterations * num_rows, 1);
213 
214  Z.push_back(get_cast<standart_type>(0.0));
215  rot(1, get_data(Z) + inner_iterations-1, 1, get_data(Z) + inner_iterations, 1, C.back(), S.back());
216  }
217  }
218 
219  if (H.size() > 0){
220  // this happens when the last iteration overestimated the residual and the new iteration
221  // simply declared "converged" at iteration 0, then there is no point to adjust x
222  tpsv('U', 'N', 'N', (int) Z.size(), H, Z);
223  krylov_combine(engine, num_rows, (int) Z.size(), W, Z, x);
224  }
225  outer_iterations++;
226  outer_res = inner_res;
227  }
228 
229  return total_iterations;
230 }
