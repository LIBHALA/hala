87 template<typename hscalar,
88  typename hvectorx, typename hvectorb,
89  typename htypea, typename htypeb, typename htypec, typename htyped, typename htypef,
90  typename htypeg, typename htypeh, typename htypei, typename htypej, typename htypek, typename htypel,
91  typename htypem, typename htypen>
92 int solve_cg_core(htypea apply_preconditioner,
93  htypeb apply_operator,
94  htypen apply_operator_internal,
95  hvectorx const &b, hvectorb &x,
96  htypem scalar_make, // return a scalar, useful with auto-initialize
97  htypec scalar_division,
98  htyped scalar_move,
99  htypef vector_scale,
100  htypeg vector_add,
101  htypeh vector_sub,
102  htypei vector_sadd,
103  htypej vector_ssub,
104  htypek vector_dot,
105  htypel check_converged){
106  auto r = new_vector(b);
107  vcopy(b, r);
108 
109  auto p = new_vector(x);
110  apply_operator(x, p);
111  vector_sub(p, r); // r = b - A x
112 
113  auto Ap = new_vector(x);
114  auto z = new_vector(x);
115 
116  apply_preconditioner(r, z); // z = E^-T E^-1 r
117 
118  auto zr = scalar_make();
119  auto a = scalar_make();
120 
121  vcopy(z, p); // p = z
122 
123  vector_dot(r, z, zr); // zr = <r, z>
124 
125  int iterations = 1; // applied one Ax with preconditioner
126  bool iterate = true;
127 
128  while(iterate){
129  apply_operator_internal(p, Ap); // Ap = A * p
130  iterations++;
131 
132  auto nzr = scalar_make();
133  vector_dot(p, Ap, nzr);
134  scalar_division(zr, nzr, a); // the correction term alpha = <r, z> / <p, A p>
135 
136  vector_sadd(a, p, x); // x += a * p, x is the new solution state
137  vector_ssub(a, Ap, r); // r -= a * ap, r is the new residual
138 
139  iterate = !check_converged(iterations, r); // if not converged, keep going
140 
141  if (iterate){ // skip this part of done already
142  apply_preconditioner(r, z); // z = E^-T E^-1 r
143 
144  // keep the new zr = dot(r, z), and get p = z - beta * p, where beta = new_zr / old_zr
145  vector_dot(r, z, nzr);
146  scalar_division(nzr, zr, a); // p-correction alpha = new <r, z> / old <r, z>
147 
148  vector_scale(a, p);
149  vector_add(z, p); // p = z + p
150 
151  scalar_move(nzr, zr); // keep nzr = <z, r> so we do not recompute the dot-product
152  }
153  }
154 
155  return iterations;
156 }

180 template<class VectorLikeP, class VectorLikeI, class VectorLikeV, class VectorLikeX, class VectorLikeB>
181 int solve_cg(stop_criteria<get_precision_type<VectorLikeV>> const &stop,
182  VectorLikeP const &pntr, VectorLikeI const &indx, VectorLikeV const &vals,
183  preconditioner_noe<VectorLikeV> precon,
184  VectorLikeB const &b, VectorLikeX &x){
185 
186  check_types(vals, x, b);
187  check_types_int(pntr, indx);
188 
189  auto num_entries = get_size(pntr);
190  assert( num_entries > 1 ); // smallest case has one entry
191  assert( check_size(b, num_entries - 1) );
192 
193  if (get_size(x) < num_entries - 1){
194  force_size(num_entries - 1, x);
195  set_zero(num_entries - 1, x); // if using CUDA the entries will not be initialized
196  }
197 
198  using temp_vector = decltype( new_vector( std::declval<VectorLikeV>() ) );
199  using scalar_type = get_standard_type<VectorLikeX>;
200 
201  int nrows = get_size_int(pntr) - 1;
202 
203  auto matrix = make_sparse_matrix(nrows, pntr, indx, vals);
204  size_t bsize = sparse_gemv_buffer_size(matrix, 'N', 1.0, b, 0.0, x);
205  auto work_buffer = new_vector(x);
206  set_size(bsize, work_buffer);
207 
208  return solve_cg_core<scalar_type>(precon,
209  [&](VectorLikeX const &v, temp_vector &r)->void{
210  sparse_gemv(matrix, 'N', 1.0, v, 0.0, r, work_buffer);
211  },
212  [&](temp_vector const &v, temp_vector &r)->void{
213  sparse_gemv(matrix, 'N', 1.0, v, 0.0, r, work_buffer);
214  },
215  b, x,
216  []()->scalar_type{ return get_cast<scalar_type>(0); },
217  [](scalar_type const &num, scalar_type const &denom, scalar_type &result)->void{ result = num / denom; },
218  [](scalar_type const &src, scalar_type &dest)->void{ dest = src; },
219  [&](scalar_type const &a, temp_vector &r)->void{ scal(a, r); },
220  [&](temp_vector const &v, temp_vector &r)->void{ axpy(1.0, v, r); },
221  [&](temp_vector const &v, temp_vector &r)->void{ axpy(-1.0, v, r); },
222  [&](scalar_type const &a, temp_vector const &v, VectorLikeX &r)->void{ axpy(a, v, r); },
223  [&](scalar_type const &a, temp_vector const &v, temp_vector &r)->void{ axpy(-a, v, r); },
224  [&](temp_vector const &v, temp_vector const &r, scalar_type &a)->void{ a = dot(v, r); },
225  [&](int i, temp_vector const &r)->bool{ return ((i == stop.max_iter) || (norm2(r) < stop.tol)); }
226  );
