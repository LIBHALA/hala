79 void sparse_gemv(char trans, int M, int N,
80  FSA alpha, VectorLikeP const &pntr, VectorLikeI const &indx, VectorLikeV const &vals, VectorLikeX const &x,
81  FSB beta, VectorLikeY &&y){
82  check_set_size((hala_abs(beta) == 0.0), y, (is_n(trans)) ? M : N );
83  assert( valid::sparse_gemv(trans, M, N, get_back(pntr), pntr, indx, vals, x, y) );
84 
85  using scalar_type = get_standard_type<VectorLikeV>;
86  auto const fsalpha = get_cast<scalar_type>(alpha);
87  auto const fsbeta = get_cast<scalar_type>(beta);
88 
89  if (is_n(trans))
90  sparse_gemv_array<scalar_type, 'N'>(M, N, fsalpha, convert(pntr), convert(indx), convert(vals), convert(x), fsbeta, cconvert(y));
91  else if (is_c(trans))
92  sparse_gemv_array<scalar_type, 'C'>(M, N, fsalpha, convert(pntr), convert(indx), convert(vals), convert(x), fsbeta, cconvert(y));
93  else
94  sparse_gemv_array<scalar_type, 'T'>(M, N, fsalpha, convert(pntr), convert(indx), convert(vals), convert(x), fsbeta, cconvert(y));
95 }
