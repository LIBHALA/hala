87 inline void check_cuda(cublasStatus_t cublasStatus, std::string const &function_name){
88  if (cublasStatus != CUBLAS_STATUS_SUCCESS){
89  std::string message = "ERROR: cuBlas failed with code: ";
90  if (cublasStatus == CUBLAS_STATUS_NOT_INITIALIZED){
91  message += "CUBLAS_STATUS_NOT_INITIALIZED";
92  }else if (cublasStatus == CUBLAS_STATUS_ALLOC_FAILED){
93  message += "CUBLAS_STATUS_ALLOC_FAILED";
94  }else if (cublasStatus == CUBLAS_STATUS_INVALID_VALUE){
95  message += "CUBLAS_STATUS_INVALID_VALUE";
96  }else if (cublasStatus == CUBLAS_STATUS_ARCH_MISMATCH){
97  message += "CUBLAS_STATUS_ARCH_MISMATCH";
98  }else if (cublasStatus == CUBLAS_STATUS_MAPPING_ERROR){
99  message += "CUBLAS_STATUS_MAPPING_ERROR";
100  }else if (cublasStatus == CUBLAS_STATUS_EXECUTION_FAILED){
101  message += "CUBLAS_STATUS_EXECUTION_FAILED";
102  }else if (cublasStatus == CUBLAS_STATUS_INTERNAL_ERROR){
103  message += "CUBLAS_STATUS_INTERNAL_ERROR";
104  }else if (cublasStatus == CUBLAS_STATUS_NOT_SUPPORTED){
105  message += "CUBLAS_STATUS_NOT_SUPPORTED";
106  }else if (cublasStatus == CUBLAS_STATUS_LICENSE_ERROR){
107  message += "CUBLAS_STATUS_LICENSE_ERROR";
108  }else{
109  message += "UNKNOWN";
110  }
111  throw std::runtime_error(message + " at " + function_name);
112  }
113 }
