229 template<bool conjugate, typename T>
230 inline T dot_array(int N, T const x[], int incx, T const y[], int incy){
231  T sum = get_cast<T>(0.0);
232  if ((incx == 1) && (incy == 1)){
233  if __HALA_CONSTEXPR_IF__ (conjugate){
234  for(int i=0; i<N; i++) sum += cconj(x[i]) * y[i];
235  }else{
236  for(int i=0; i<N; i++) sum += x[i] * y[i];
237  }
238  }else{
239  if __HALA_CONSTEXPR_IF__ (conjugate){
240  for(int i=0; i<N; i++) sum += cconj(x[i * incx]) * y[i * incy];
241  }else{
242  for(int i=0; i<N; i++) sum += x[i * incx] * y[i * incy];
243  }
244  }
245  return sum;
246 }
