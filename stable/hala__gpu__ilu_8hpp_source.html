123  template<class VectorLikeX, class VectorLikeR>
124  size_t buffer_size(VectorLikeX const &x, VectorLikeR &&r, int num_rhs) const{
125  gpu_pntr<host_pntr> hold(engine());
126  if (num_rhs == 1){

129  }else{

131  }

147  template<class VectorLikeX, class VectorLikeR, class VectorLikeT>
148  void apply(VectorLikeX const &x, VectorLikeR &&r, int num_rhs, VectorLikeT &&temp_buff) const{
149  check_types(x, r);
150  check_set_size(assume_output, r, num_rhs, num_rows);
151  gpu_pntr<host_pntr> hold(engine());
152 
153  if (num_rhs == 1){
154  auto tmp = new_vector(engine(), r);
155  lower->trsv('N', 1.0, x, tmp, temp_buff);
156  upper->trsv('N', 1.0, tmp, r, temp_buff);
157  }else{
158  vcopy(engine(), x, r);
159  lower->trsm('N', 'N', num_rhs, 1.0, r, num_rows, temp_buff);
160  upper->trsm('N', 'N', num_rhs, 1.0, r, num_rows, temp_buff);
161  }
162  }
