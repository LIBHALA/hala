279  template<typename FPA, class VectorLikeB, class VectorLikeT>
280  void trsm(char transa, char transb, int nrhs, FPA alpha, VectorLikeB &&B, int ldb, VectorLikeT &&) const{
281  check_types(rvals, B);
282  valid::default_ld(is_n(transb), nrows, nrhs, ldb);
283  assert( valid::sparse_trsm(transa, transb, nrhs, *this, B, ldb) );
284 
285  using standard_type = get_standard_type<std::vector<value_type>>;
286  auto bdata = get_standard_data(B);
287 
288  if (is_n(transb)){ // non-transpose B
289  std::vector<standard_type> x(nrows);
290  for(int i=0; i<nrhs; i++){
291  vcopy(nrows, &bdata[hala_size(ldb, i)], 1, x, 1);
292  trsv(transa, alpha, x, &bdata[hala_size(ldb, i)]);
293  }
294  }else{ // the conjugate operation cancels out on both sides of the equation
295  std::vector<standard_type> x(nrows), btemp(nrows);
296  for(int i=0; i<nrhs; i++){
297  vcopy(nrows, &bdata[i], ldb, x, 1);
298  trsv(transa, alpha, x, btemp);
299  vcopy(nrows, btemp, 1, &bdata[i], ldb);
300  }
301  }
302  }
304  template<typename FPA, class VectorLikeB>
305  void trsm(char transa, char transb, int nrhs, FPA alpha, VectorLikeB &&B, int ldb = -1) const{
306  valid::default_ld(is_n(transb), nrows, nrhs, ldb);
307  trsm(transa, transb, nrhs, alpha, B, ldb,
308  std::vector<value_type>(trsm_buffer_size(transa, transb, nrhs, alpha, B, ldb)));
309  }
